# ToggleSwitch

## Utilisation

### Simple

<Playground>
  {() => {
    const [isToggled, toggle] = useState(true)
    return (
      <ToggleSwitch
        id="toggle1"
        name="toggle1"
        onChange={() => toggle(!isToggled)}
        toggled={isToggled}
      />
    )
  }}
</Playground>
